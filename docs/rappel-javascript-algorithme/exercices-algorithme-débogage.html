function fonction8(tableau) {
  let a = tableau[0];
  let b = tableau[0];
  for (let i = 1; i < tableau.length; i++) {
    if (tableau[i] < a) {
      a = tableau[i];
    }
    if (tableau[i] > b) {
      b = tableau[i];
    }
  }

}
